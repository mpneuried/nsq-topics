49    set: ( key, value )=>
50      if not key?
51        return
52      
53      if key is "set"
54        return
55      
56      if _isObject( key )
57        for _k, _v of key when key.hasOwnProperty( _k )
58          @set( _k, _v )
59        return
60      if _isObject( @[ key ] ) and _isObject( value ) and not _isArray( value ) and not _isFunction( value )
61        @[ key ] = _assignIn( true, {}, @[ key ], value )
62      else
63        @[ key ] = value
